# How to refresh an Angular Component

Sometimes we need to refresh our Angular Component only, not the entire page.
For this reason, we are going to learn how we can do this.


## Create a component


For this example, we are going to use this simple component.

In this example, im going to add a Spotify Podcast Frame <br/>(this component is in the folder '/components'): <br/><br/>

Component:
```jsx
import { Component } from '@angular/core';
import { Router } from '@angular/router';
import { HttpService } from '../../services/httpService.service';


@Component({
  selector: 'productsGridComponent',
  templateUrl: './productsGrid.component.html',
  providers: [], //Utilizamos el servicio aqui
  styleUrls: ['./productsGrid.component.scss']
})



export class ProductsGridComponent {
  posts: any;

  
  constructor(private httpService: HttpService, private router: Router) { }

  ngOnInit() {
    this.getProducts()
  }
  
  async getProducts(){
    this.httpService.getPosts().subscribe(
      (response) => { this.posts = response; },
      (error) => { console.log(error); }
    ); 
  }
  
  async deleteProduct(uid: string){
    this.httpService.deleteProduct(uid).subscribe(
      (response) => { console.log("Product dropped"); },
      (error) => { console.log(error); }
    ); 
      
    await this.getProducts()
    this.reloadComponent()
    // this.ngOnInit();
  }
  
  reloadComponent() {
    let currentUrl = this.router.url;
        this.router.routeReuseStrategy.shouldReuseRoute = () => false;
        this.router.onSameUrlNavigation = 'reload';
        this.router.navigate([currentUrl]);
    }

}

```
